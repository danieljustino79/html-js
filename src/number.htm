Number

numero primitivo, new Number(), decimal e hexa. 
var numero = 123;
var numeroDT = typeof(numero); //--number
var numeroObjeto = new Number(456);
var numeroObjetoDT = typeof(numeroObjeto); //--object

var decimal = 10.45;
var decimalDT = typeof(decimal); //--number

var hexa = 0xFF; 
var hexaDT = typeof(hexa); //--number

principais métodos e metodo nativo:
IsNaN() - metodo nativo Not-a-Number
toFixed(precision) - arrendodamento e preenchimento
valueOf() - retorna o numero de um objeto
parseInt() - decimal para inteiro

var teste = 'ok';

 var v1 = 12.7; 
console.log(v1.toFixed(2)) //--12.70

var v2 = new Number(2);
console.log(typeof(v2))           //--object
console.log(typeof(v2.valueOf())) //--number

var v3 = 12.8;
var v4 = parseInt(v3);
console.log(v4) //--12
